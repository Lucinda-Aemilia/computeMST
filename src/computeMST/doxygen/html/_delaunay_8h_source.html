194  void AddPoint(Point p)
195  {
196  points.push_back(p);
197  int pi = points.size() - 1;
198 
199  std::vector<Tptr> bad_triangles;
200 
201  // For now I am just doing a naive search
202  // I hope to replace this one day with something different
203  for (auto T : triangles)
204  {
205  if (CircumcircleContains(T, p))
206  bad_triangles.push_back(T);
207  }
208 
209  // Find the boundary of the bad triangles
210  std::vector<Edge> boundary = GetBoundary(bad_triangles);
211 
212  // Remove all the bad triangles from the list of triangles
213  for (auto T : bad_triangles)
214  triangles.erase(std::remove(triangles.begin(), triangles.end(), T), triangles.end());
215 
216  // Retriangle the hole just created
217  std::vector<Tptr> new_triangles;
218  for (auto edge : boundary)
219  {
220  int a = std::get<0>(edge);
221  int b = std::get<1>(edge);
222 
223  auto T = std::make_shared<Triangle>(pi, a, b);
224 
225  T->n[0] = std::get<2>(edge); // To neighbour
226 
227  if (std::get<2>(edge))
228  T->n[0]->SetEdge(Edge(b, a, nullptr), T); // From neighbour
229 
230  new_triangles.push_back(T);
231  }
232 
233  // Link the new triangles to each other
234  int N = new_triangles.size();
235  for (int i = 0; i < N; i++)
236  {
237  new_triangles[i]->n[2] = new_triangles[((i-1) % N + N) % N];
238  new_triangles[i]->n[1] = new_triangles[(i+1) % N];
239  }
240 
241  triangles.reserve(triangles.size() + distance(new_triangles.begin(), new_triangles.end()));
242  triangles.insert(triangles.end(), new_triangles.begin(), new_triangles.end());
243  }

247  // Check whether the circumcirlce of T contains p
248  bool CircumcircleContains(Tptr T, Point p)
249  {
250  Point a = points[T->v[0]] - points[T->v[2]];
251  Point b = points[T->v[1]] - points[T->v[2]];
252 
253  // Ref: https://en.wikipedia.org/wiki/Circumscribed_circle#Circumcircle_equations
254  Point z = cross(a,b);
255  Point p0 = cross(dot(a,a)*b-dot(b,b)*a, z)*(0.5/dot(z,z)) + points[T->v[2]];
256 
257  double r2 = 0.25*dot(a, a)*dot(b,b)*dot(a-b, a-b)/dot(z, z);
258 
259  return (dot(p-p0, p-p0) <= r2);
260  }

262  std::vector<Edge> GetBoundary(std::vector<Tptr> bad_triangles)
263  {
264 
265  // Start with a triangle at random
266  Tptr T = bad_triangles[0];
267  int edge = 0;
268 
269  // Create empty boundary list
270  std::vector<Edge> boundary;
271 
272  while (true)
273  {
274  if (boundary.size() > 1)
275  if (boundary.front() == boundary.back())
276  break;
277 
278  // Check if this edge is shared with a triangle in bad_triangles.
279  if (std::find(bad_triangles.begin(), bad_triangles.end(), T->n[edge]) != bad_triangles.end())
280  {
281  // If so set current triangle
282  Tptr last = T;
283  T = T->n[edge];
284 
285  int pos = find(T->n.begin(), T->n.end(), last) - T->n.begin();
286  edge = (pos + 1) % 3;
287  }
288 
289  else // Found an edge that is on the boundary
290  {
291  // Add to list
292 
293  Edge new_edge(T->v[(edge+1)%3], T->v[(edge+2)%3], T->n[edge]);
294  boundary.push_back( new_edge );
295  edge = (edge + 1) % 3;
296  }
297  }
298 
299  boundary.pop_back();
300  return boundary;
301  }
